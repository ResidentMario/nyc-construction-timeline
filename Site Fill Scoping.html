def simple_date_map(val):
    try:
        return np.datetime64(arrow.get(val, "MM/DD/YYYY").datetime)
    except:
        return np.nan

def less_simple_date_map(val):
    try:
        return np.datetime64(arrow.get(val.split(" ")[0], "MM/DD/YYYY").datetime)
    except:
        return np.nan
